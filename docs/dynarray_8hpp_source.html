118 template <class T> bool dynarray<T>::insert_element(size_t index, T element)
119 {
120  bool retval = false;
121 
122  // Note: Index can be 0 if inserting new lements at the beginning
123  // Note: Index can be == m_logical_size if inserting elements at the end
124  if (index <= m_logical_size)
125  {
126  // Index is valid
127  if (m_logical_size == m_capacity)
128  {
129  // The logical size of the backing array has reached its capacity.
130  // Resize the backing array to 2x its current capacity
131  // (or 1 if its current capacity is 0) and copy the old contents to
132  // the
133  // new array.
134  size_t new_capacity = CALCULATE_NEW_CAPACITY(m_capacity);
135  T *new_array = new T[new_capacity]{};
136 
137  // Allocation of new array failed. Return failure code
138  if (new_array == NULL)
139  {
140  return false;
141  }
142 
143  // Copy the rest of the items to the new_array
144  for (size_t i = 0, j = 0; i < m_capacity; ++i, ++j)
145  {
146  if (i == index)
147  {
148  // Skip the index for the new element. The new element will
149  // be copied to this location.
150  ++j;
151  }
152  new_array[j] = mp_array[i];
153  }
154 
155  delete[] mp_array; // Deallocate old array
156  mp_array = new_array; // Assign new array
157 
158  mp_array[index] = element; // Store the new element at the index
159  m_logical_size += 1; // Increment the logical size
160 
161  m_capacity =
162  new_capacity; // Update the capacity of the backing array
163  }
164  else
165  {
166  T temp = element; // Store the element in the temporary storage
167  m_logical_size += 1; // Increment the size by 1
168 
169  // Move each element to the right
170  for (size_t i = index; i < m_logical_size; ++i)
171  {
172  std::swap(mp_array[i], temp);
173  }
174  }
175  retval = true;
176  }
177 
178  return retval;
179 }

185 template <class T> bool dynarray<T>::delete_element(size_t index)
186 {
187  bool retval = false;
188 
189  // Ensure that the array is valid and the size and index are within range
190  if (mp_array && m_logical_size > 0 && index < m_logical_size)
191  {
192  mp_array[index] = {0}; // Clear existing element
193  m_logical_size -= 1; // Decrement the size of the backing array by 1
194 
195  // Sift all elements past the deleted element to the left
196  for (size_t i = index; i < m_logical_size; ++i)
197  {
198  std::swap(mp_array[i], mp_array[i + 1]);
199  }
200 
201  retval = true;
202  }
203  return retval;
204 }
